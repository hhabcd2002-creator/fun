function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheets()[0]; var data = JSON.parse(e.postData.contents); var rows = sheet.getDataRange().getValues(); // --- Sign Up (အကောင့်သစ်ဖွင့်ခြင်း) --- if (data.action == "signup") { for (var i = 1; i < rows.length; i++) { if (rows[i][0] == data.username) return ContentService.createTextOutput("User Exists"); } sheet.appendRow([data.username, data.password, 0]); // username, password, gold return ContentService.createTextOutput("Success"); } // --- Sign In (အကောင့်ဝင်ခြင်း) --- if (data.action == "signin") { for (var i = 1; i < rows.length; i++) { if (rows[i][0] == data.username && rows[i][1] == data.password) { var userObj = { username: rows[i][0], gold: rows[i][2] }; return ContentService.createTextOutput(JSON.stringify(userObj)); } } return ContentService.createTextOutput("Wrong Credentials"); } // --- Update Gold (Gold ပမာဏ သိမ်းဆည်းခြင်း) --- if (data.action == "updateGold") { for (var i = 1; i < rows.length; i++) { if (rows[i][0] == data.username) { sheet.getRange(i + 1, 3).setValue(data.gold); return ContentService.createTextOutput("Updated"); } } } }
